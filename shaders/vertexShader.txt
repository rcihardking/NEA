#version 420 core
layout (location = 0) in vec3 inPos;
layout (location = 1) in vec2 inTex;
layout (location = 2) in vec3 inNor;
  
out vec2 UV;
out vec3 normal;

uniform mat4 scale;
uniform mat4 rotation;
uniform mat4 translation;
uniform mat4 proj;

void main()
{
    //mat4 proj = mat4(1.0);
    gl_Position = proj * translation * rotation * scale * vec4(inPos, 1.0);
    UV = inTex;
    normal = inNor;
}
